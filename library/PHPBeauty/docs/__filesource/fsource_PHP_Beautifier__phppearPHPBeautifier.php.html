  561.         public function process() {
    562. 

  562.             $this->resetProperties();
    563. 

  563.             $this->aTokens = token_get_all($this->sText);
    564. 

  564.             $this->aOut = array();
    565. 

  565.             $iTotal = count($this->aTokens);
    566. 

  566.             $iPrevAssoc = false;
    567. 

  567.             for ($this->iCount = 0;$this->iCount<$iTotal;$this->iCount++) {
    568. 

  568.                 $aCurrentToken = $this->aTokens[$this->iCount];
    569. 

  569.                 if (is_string($aCurrentToken)) {
    570. 

  570.                     $aCurrentToken = array(
    571. 

  571.                         0=>$aCurrentToken,
    572. 

  572.                         1=>$aCurrentToken
    573. 

  573.                     );
    574. 

  574.                 }
    575. 

  575.  
    576. 

  576.                 $this->controlToken($aCurrentToken);
    577. 

  577.                 $iFirstOut = count($this->aOut); //5
    578. 

  578.                 $bError = false;
    579. 

  579.                 if ($this->bBeautify) {
    580. 

  580.                     foreach($this->aFilters as $oFilter) {
    581. 

  581.                         $bError = true;
    582. 

  582.                         if ($oFilter->handleToken($aCurrentToken) !== FALSE) {
    583. 

  583.                             $bError = false;
    584. 

  584.                             break;
    585. 

  585.                         }
    586. 

  586.                     }
    587. 

  587.                 } else {
    588. 

  588.                     $this->add($aCurrentToken[1]);
    589. 

  589.                 }
    590. 

  590.                 $this->controlTokenPost($aCurrentToken);
    591. 

  591.                 $iLastOut = count($this->aOut);
    592. 

  592.                 // set the assoc
    593. 

  593.                 if (($iLastOut-$iFirstOut) >0) {
    594. 

  594.                     $this->aAssocs[$this->iCount] = array(
    595. 

  595.                         'offset'=>$iFirstOut
    596. 

  596.                     );
    597. 

  597.                     if ($iPrevAssoc !== FALSE) {
    598. 

  598.                         $this->aAssocs[$iPrevAssoc]['length'] = $iFirstOut-$this->aAssocs[$iPrevAssoc]['offset'];
    599. 

  599.                     }
    600. 

  600.                     $iPrevAssoc = $this->iCount;
    601. 

  601.                 }
    602. 

  602.                 if ($bError) {
    603. 

  603.                     throw(new Exception("Can'process token: ".var_dump($aCurrentToken)));
    604. 

  604.                 }
    605. 

  605.             } // ~for
    606. 

  606.             // generate the last assoc
    607. 

  607.             $this->aAssocs[$iPrevAssoc]['length'] = (count($this->aOut) -1) -$this->aAssocs[$iPrevAssoc]['offset'];
    608. 

  608.             // Post-processing
    609. 

  609.             foreach($this->aFilters as $oFilter) {
    610. 

  610.                 $oFilter->postProcess();
    611. 

  611.             }
    612. 

  612.             return true;
    613. 

  613.         }

  657.         /**
    658. 

  658.         * Assign value for some variables with the information of the token
    659. 

  659.         * @param array current token
    660. 

  660.         */
    661. 

  661.         private function controlToken($aCurrentToken) {
    662. 

  662.             // is a control structure opener?
    663. 

  663.             if (in_array($aCurrentToken[0], $this->aControlStructures)) {
    664. 

  664.                 $this->pushControlSeq($aCurrentToken);
    665. 

  665.                 $this->iControlLast = $aCurrentToken[0];
    666. 

  666.             }
    667. 

  667.             // is a control structure closer?
    668. 

  668.             if (in_array($aCurrentToken[0], $this->aControlStructuresEnd)) {
    669. 

  669.                 $this->popControlSeq();
    670. 

  670.             }
    671. 

  671.             switch ($aCurrentToken[0]) {
    672. 

  672.                 case T_COMMENT:
    673. 

  673.                     // callback?
    674. 

  674.                     if (preg_match("/\/\/\s*(.*?)->((.*)\((.*)\))/", $aCurrentToken[1], $aMatch)) {
    675. 

  675.                         try {
    676. 

  676.                             if (stristr('php_beautifier', $aMatch[1]) and method_exists($this, $aMatch[3])) {
    677. 

  677.                                 eval ('$this->'.$aMatch[2].";");
    678. 

  678.                             } else {
    679. 

  679.                                 foreach($this->aFilters as $iIndex=>$oFilter) {
    680. 

  680.                                     if (strtolower(get_class($oFilter)) == 'php_beautifier_filter_'.strtolower($aMatch[1]) and method_exists($oFilter, $aMatch[3])) {
    681. 

  681.                                         eval ('$this->aFilters['.$iIndex.']->'.$aMatch[2].";");
    682. 

  682.                                         break;
    683. 

  683.                                     }
    684. 

  684.                                 }
    685. 

  685.                             }
    686. 

  686.                         }
    687. 

  687.                         catch(Exception $oExp) {
    688. 

  688.                         }
    689. 

  689.                     }
    690. 

  690.                 break;
    691. 

  691.                 case T_FUNCTION:
    692. 

  692.                     $this->setMode('function');
    693. 

  693.                 break;
    694. 

  694.                 case T_ARRAY:
    695. 

  695.                     $this->iArray++;
    696. 

  696.                 break;
    697. 

  697.                 case T_WHITESPACE:
    698. 

  698.                     $this->currentWhitespace = $aCurrentToken[1];
    699. 

  699.                 break;
    700. 

  700.                 case '{':
    701. 

  701.                     if ($this->getPreviousTokenConstant() == T_VARIABLE or ($this->getPreviousTokenConstant() == T_STRING and $this->getPreviousTokenConstant(2) == T_OBJECT_OPERATOR)) {
    702. 

  702.                         $this->setMode('string_index');
    703. 

  703.                     }
    704. 

  704.                 break;
    705. 

  705.                 case '}':
    706. 

  706.                     if ($this->iVerbose) {
    707. 

  707.                         echo 'end bracket:'.$this->getPreviousTokenContent() ."\n";
    708. 

  708.                     }
    709. 

  709.                     if ($this->getPreviousTokenContent() == ';' or $this->getPreviousTokenContent() == '}' or $this->getPreviousTokenContent() == '{') {
    710. 

  710.                         $this->popControlSeq();
    711. 

  711.                     }
    712. 

  712.                 break;
    713. 

  713.                 case '(':
    714. 

  714.                     $this->iParenthesis++;
    715. 

  715.                     $this->pushControlParenthesis();
    716. 

  716.                 break;
    717. 

  717.                 case ')':
    718. 

  718.                     $this->iParenthesis--;
    719. 

  719.                 break;
    720. 

  720.                 case '?':
    721. 

  721.                     $this->setMode('ternary_operator');
    722. 

  722.                 break;
    723. 

  723.             }
    724. 

  724.             if ($this->getTokenFunction($aCurrentToken[0]) == 't_include') {
    725. 

  725.                 $this->setMode('include');
    726. 

  726.             }
    727. 

  727.         }
