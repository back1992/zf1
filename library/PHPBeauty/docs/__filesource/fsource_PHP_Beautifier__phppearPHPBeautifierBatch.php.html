  190.         private function getWithHeader($sFile) {
    191. 

  191.             $sError=false;
    192. 

  192.             try {
    193. 

  193.                 $this->oBeaut->setInputFile($sFile);
    194. 

  194.                 $this->oBeaut->process();
    195. 

  195.             } catch (Exception $oExp) {
    196. 

  196.                 $sError=$oExp->getMessage();
    197. 

  197.             }
    198. 

  198.             $sHeader='- BEGIN OF '.$sFile.' -'.$this->oBeaut->sNewLine;
    199. 

  199.             $sLine=str_repeat('-',strlen($sHeader)-1).$this->oBeaut->sNewLine;
    200. 

  200.             $sEnd='- END OF '.$sFile.str_repeat(' ',strlen($sHeader)-strlen($sFile)-12).' -'.$this->oBeaut->sNewLine;
    201. 

  201.             $sText=$sLine.$sHeader.$sLine.$this->oBeaut->sNewLine;
    202. 

  202.             $sText.=($sError)?$sError:$this->oBeaut->get();
    203. 

  203.             $sText.=$this->oBeaut->sNewLine.$sLine.$sEnd.$sLine.$this->oBeaut->sNewLine;
    204. 

  204.             return $sText;
    205. 

  205.         }

  251.         private function getSavePath($aFiles, $sPath = './') 
    252. 

  252.         {
    253. 

  253.             $sPath = $this->normalizeDir($sPath);
    254. 

  254.             // get the lowest denominator..
    255. 

  255.             $sPrevious = '';
    256. 

  256.             $iCut = 0;
    257. 

  257.             foreach($aFiles as $sFile) {
    258. 

  258.                 if (!$sPrevious) {
    259. 

  259.                     $sPrevious = dirname($sFile);
    260. 

  260.                     continue;
    261. 

  261.                 }
    262. 

  262.                 for ($x = 0;$x<strlen($sPrevious);$x++) {
    263. 

  263.                     $sDir=dirname($sFile);
    264. 

  264.                     if (strlen($sDir)<$x+1) {
    265. 

  265.                         $sPrevious='';
    266. 

  266.                         break(2);
    267. 

  267.                     } elseif ($sDir{$x} != $sPrevious{$x}) {
    268. 

  268.                         $sPrevious = substr($sDir, 0, $x);
    269. 

  269.                         continue;
    270. 

  270.                     }
    271. 

  271.                 }
    272. 

  272.             }
    273. 

  273.                 $iCut = strlen($sPrevious);
    274. 

  274.             $aPathsOut = array();
    275. 

  275.             foreach($aFiles as $sFile) {
    276. 

  276.                 $sFileOut=preg_replace("/^(\w:\/|\.\/|\/)/","",substr($sFile,$iCut));
    277. 

  277.                 $aPathsOut[] = $sPath.$sFileOut;
    278. 

  278.             }
    279. 

  279.             return $aPathsOut;
    280. 

  280.         }
